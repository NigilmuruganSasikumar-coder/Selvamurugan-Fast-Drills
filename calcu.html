window.onload = function () { // ========================== // Load Saved Data // ========================== const data = JSON.parse(localStorage.getItem("quotationData")); if (!data) { alert("No quotation data found."); return; } // ========================== // Customer Details // ========================== document.getElementById("pdfCustomerName").textContent = data.customerName || "-"; document.getElementById("pdfCustomerPhone").textContent = data.customerPhone || "-"; // ========================== // Drilling Details // ========================== document.getElementById("pdfMode").textContent = data.mode.toUpperCase(); document.getElementById("pdfDepth").textContent = data.depth + " ft"; document.getElementById("pdfRate").textContent = "₹" + data.baseRate + " / ft"; document.getElementById("pdfGST").textContent = data.gst ? "ON" : "OFF"; // ========================== // Date // ========================== const today = new Date(); const formattedDate = String(today.getDate()).padStart(2, "0") + "/" + String(today.getMonth() + 1).padStart(2, "0") + "/" + today.getFullYear(); document.getElementById("pdfDate").textContent = formattedDate; // ========================== // Values // ========================== const depth = Number(data.depth); const baseRate = Number(data.baseRate); const casing7Feet = Number(data.casing7); const casing7Rate = Number(data.casing7Rate); const casing10Feet = Number(data.casing10); const casing10Rate = Number(data.casing10Rate); const reboreFeet = Number(data.reboreFeet); const reboreRate = Number(data.reboreRate); let drillingTotal = 0; let start = 1; let currentRate = baseRate; let addValue = 5; const tbody = document.getElementById("pdfBreakdown"); tbody.innerHTML = ""; // ========================== // Rebore Section // ========================== if (data.mode === "rebore") { const reboreCost = reboreFeet * reboreRate; drillingTotal += reboreCost; tbody.innerHTML += ` 001 - ${reboreFeet} ft ₹${reboreRate}/ft ₹${reboreCost.toLocaleString()} `; start = reboreFeet + 1; if (reboreFeet > 300) { let slab = Math.floor((reboreFeet - 1) / 100); currentRate = baseRate; addValue = 5; for (let s = 3; s <= slab; s++) { currentRate += addValue; if (addValue === 5) { addValue = 10; } else { addValue += 10; } } } } // Save all variables for Part 2 window.quote = { data, depth, casing7Feet, casing7Rate, casing10Feet, casing10Rate, drillingTotal, currentRate, addValue, start, tbody }; // ========================== // Drilling Breakdown // ========================== while (start <= depth) { let end; if (start <= 300) { end = Math.min(300, depth); } else { end = Math.min( Math.ceil(start / 100) * 100, depth ); } const feet = end - start + 1; const slabCost = feet * currentRate; drillingTotal += slabCost; tbody.innerHTML += ` ${String(start).padStart(3, "0")} - ${end} ft ₹${currentRate}/ft ₹${slabCost.toLocaleString()} `; start = end + 1; if (end < 1000) { currentRate += addValue; if (addValue === 5) { addValue = 10; } else { addValue += 10; } } else { currentRate += 100; } } // ========================== // PVC Cost // ========================== const pvc7Cost = casing7Feet * casing7Rate; const pvc10Cost = casing10Feet * casing10Rate; // ========================== // Sub Total // ========================== const subTotal = drillingTotal + pvc7Cost + pvc10Cost; // ========================== // GST // ========================== let gstAmount = 0; if (data.gst) { gstAmount = Math.round(subTotal * 0.18); } // ========================== // Grand Total // ========================== const grandTotal = subTotal + gstAmount; // ========================== // Fill Summary // ========================== document.getElementById("pdfDrillingTotal").textContent = "₹" + drillingTotal.toLocaleString(); document.getElementById("sumDrilling").textContent = "₹" + drillingTotal.toLocaleString(); document.getElementById("sumPVC7").textContent = "₹" + pvc7Cost.toLocaleString(); document.getElementById("sumPVC10").textContent = "₹" + pvc10Cost.toLocaleString(); document.getElementById("sumGST").textContent = "₹" + gstAmount.toLocaleString(); document.getElementById("sumGrand").textContent = "₹" + grandTotal.toLocaleString(); const params = new URLSearchParams(window.location.search); if (params.get("download") === "true") { setTimeout(async () => { await autoDownloadPDF(); window.close(); }, 800); } }; // ========================================== // Download PDF // ========================================== async function autoDownloadPDF() { const { jsPDF } = window.jspdf; const element = document.getElementById("quotation"); const canvas = await html2canvas(element, { scale: 2, useCORS: true }); const imgData = canvas.toDataURL("image/png"); const pdf = new jsPDF("p", "mm", "a4"); const pageWidth = 210; const pageHeight = 297; const imgWidth = pageWidth; const imgHeight = canvas.height * imgWidth / canvas.width; pdf.addImage( imgData, "PNG", 0, 0, imgWidth, imgHeight ); pdf.save("Selvamurugan_Fast_Drills_Quotation.pdf"); }
